	/** Global Variables **/
	
	/*
	You have to be careful about using Global Variables 
	as object or function names because you never know 
	when that name might be used again in a library or 
	a different source of code. 
	*/
	
	var names = ['zero', 'one', 'two',
		'three', 'four', five', 'six',
		'seven', 'eight', 'nine'];
	
	var digit_name = function (n) {
		return names[n];
	};
		
	alert(digit_name(3));  // 'three'
	
	/*
	Here the names variable could be a global variable used
	in other functions or libraries
	*/
	
	/** Global Variables Fixed **/
	
	var digit_name = function () {
		var names = ['zero', 'one', 'two',
		'three', 'four', five', 'six',
		'seven', 'eight', 'nine'];
		
		return function (n) {
			return names[n];
		};
	}();
	alert(digit_name(3)); // 'three'
	
	/*
	Here the function returns another function then the outer function
	gets returned immediately. This prevents any issues with other "names" 
	variables because it only accesses the names variable in the digit_name function. 
	*/
	
	/** Style Isn't Subjective **/
	
	//Running the code below will cause a SILENT ERROR
	
	return
	{
		ok: false
	};
	
	/*
	Javascript will add a semicolon after return causing 
	it to return noothing, no object or error just an empty return.
	*/
	
	//The code below will cause no errors and is the correct style for JavaScript
	
	return {
		ok: true
	};
